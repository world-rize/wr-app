void toggleSpeed() {
  final _index = _playbackSpeeds.indexOf(speed);
  final _next = _playbackSpeeds[(_index + 1) % _playbackSpeeds.length];
  _player.fixedPlayer.setPlaybackRate(playbackRate: _next);
  speed = _next;
  notifyListeners();
}
