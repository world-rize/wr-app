Future<User> resetTestCount({
  @required User user,
}) async {
  user.testLimitCount = 3;
  final res =
      await _userRepository.updateUser(UpdateUserRequestDto(user: user));

  if (!(res?.success ?? false)) {
    throw Exception('failed');
  }

  return user;
}
